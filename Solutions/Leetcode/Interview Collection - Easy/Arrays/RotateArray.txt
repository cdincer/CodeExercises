class Solution {
    public void rotate(int[] nums, int k) {
        int temp=0;
        for(int i=0;i<nums.length-k;i++)
        {
            temp=nums[i+k];
            nums[i+k]=nums[i];

            if(i+k+k>nums.length)
            {
            nums[i+k+k-nums.length]=temp;
            }
            else
            {
                  nums[i+k+k-1]=temp;
            }


        }


    }
}

//First transition is where we move the first element,x number of steps,thats completed
//Second transition the late item.Last spot.
